let runLengthEncoding = function(str){
  let arr = [];
  let count = 1;
  for(let i = 0; i < str.length; i++) {
    if(str[i] === str[i + 1]) {
      count++;
    } else {

    }
  }

function unluckyDays(year){
  let count = 0;
  for(let i = 0; i < 12; i++) {
    let date = new Date(year, i, 13);

  }

let validWord = function(dictionary, word) {
  let words = dictionary.join('|');
  const regex = new RegExp('^(' + words + ')*$');
  return word.match(regex) ? true : false;
};
